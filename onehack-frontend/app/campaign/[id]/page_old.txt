"use client";
import {
  useSuiClientQuery,
  useSignAndExecuteTransaction,
  useSuiClient,
  useCurrentAccount,
} from "@mysten/dapp-kit";
import { Transaction } from "@mysten/sui/transactions";
import { PACKAGE_ID, MODULE, COIN_TYPE } from "@/config";
import { useState, useMemo } from "react";
import { useParams } from "next/navigation";
import Link from "next/link";

export default function CampaignDetails() {


  const { id } = useParams();

  
  const client = useSuiClient();
  const account = useCurrentAccount();
  const { mutate: signAndExecute } = useSignAndExecuteTransaction();
  const [amount, setAmount] = useState("");
  const [loading, setLoading] = useState(false);
  const [copied, setCopied] = useState(false);

  const shortHash = (hash: string) => {
    if (!hash) return "Unknown";
    return `${hash.slice(0, 6)}...${hash.slice(-4)}`;
  };

  const { data: objectData, refetch } = useSuiClientQuery("getObject", {
    id: id as string,
    options: { showContent: true },
  });

  const { data: eventsData } = useSuiClientQuery("queryEvents", {
    query: { MoveModule: { package: PACKAGE_ID, module: MODULE } },
    limit: 100,
    order: "descending",
  });

  const creationEvent = eventsData?.data.find(
    (e: any) => e.type.includes("::CampaignCreated") && (e.parsedJson as any)?.campaign_id === id
  );

  console.log("Events Data:", eventsData);
  const createdDate = creationEvent
    ? new Date(Number(creationEvent.timestampMs)).toLocaleDateString()
    : "Just now";

  const { topDonors, recentDonations } = useMemo(() => {
    if (!eventsData) return { topDonors: [], recentDonations: [] };

    const donations = eventsData.data.filter(
      (e: any) =>
        (e.parsedJson as any)?.campaign_id === id &&
        e.type.includes("::DonationReceived")
    );

    const donorMap: Record<string, number> = {};
    donations.forEach((e: any) => {
      const { donor, amount_donated } = e.parsedJson;
      donorMap[donor] = (donorMap[donor] || 0) + Number(amount_donated);
    });

    const sortedTopDonors = Object.entries(donorMap)
      .map(([donor, total]) => ({ donor, total }))
      .sort((a, b) => b.total - a.total);

    return { topDonors: sortedTopDonors, recentDonations: donations };
  }, [eventsData, id]);

  const handleShare = () => {
    const url = window.location.href;
    navigator.clipboard.writeText(url).then(() => {
      setCopied(true);
      setTimeout(() => setCopied(false), 2000);
    });
  };

  const donate = async () => {
    if (!account) return alert("Please connect wallet first");
    if (!amount || parseFloat(amount) <= 0) return alert("Please enter a valid amount");

    setLoading(true);

    try {
      const tx = new Transaction();
      const decimals = 1_000_000_000;
      const amountInMist = Math.floor(parseFloat(amount) * decimals);
      const GAS_BUFFER = 50_000_000; 
      const totalNeeded = amountInMist + GAS_BUFFER;

      const { data: coins } = await client.getCoins({
        owner: account.address,
        coinType: "0x2::oct::OCT",
        limit: 50,
      });

      if (coins.length === 0) throw new Error("No OCT coins found");

      let currentBalance = 0;
      const paymentCoins = [];

      for (const coin of coins) {
        paymentCoins.push({
          objectId: coin.coinObjectId,
          version: coin.version,
          digest: coin.digest,
        });
        currentBalance += parseInt(coin.balance);
        if (currentBalance >= totalNeeded) break;
      }

      if (currentBalance < totalNeeded) {
        throw new Error(
          `Insufficient Balance. Need ${totalNeeded / decimals} OCT.`
        );
      }

      tx.setGasPayment(paymentCoins);
      tx.setGasBudget(GAS_BUFFER);

      const [coin] = tx.splitCoins(tx.gas, [tx.pure.u64(amountInMist)]);

      tx.moveCall({
        target: `${PACKAGE_ID}::${MODULE}::donate`,
        typeArguments: [COIN_TYPE],
        arguments: [tx.object(id as string), coin],
      });

      signAndExecute(
        { transaction: tx },
        {
          onSuccess: () => {
            alert("Donation Successful!");
            setLoading(false);
            setAmount("");
            setTimeout(() => refetch(), 2000);
          },
          onError: (e) => {
            console.error(e);
            alert("Tx Failed: " + e.message);
            setLoading(false);
          },
        }
      );
    } catch (e: any) {
      alert(e.message);
      setLoading(false);
    }
  };

  if (!objectData?.data)
    return (
        <div className="flex h-screen items-center justify-center space-x-2 bg-gray-950">
            <div className="w-4 h-4 bg-green-500 rounded-full animate-bounce"></div>
            <div className="w-4 h-4 bg-green-500 rounded-full animate-bounce delay-75"></div>
            <div className="w-4 h-4 bg-green-500 rounded-full animate-bounce delay-150"></div>
        </div>
    );

  const fields = (objectData.data.content as any).fields;
  const raisedAmount = fields.raised / 1e9;
  const targetAmount = fields.target / 1e9;
  const progressPercent = Math.min(100, (fields.raised / fields.target) * 100);

  return (
    <div className="min-h-screen bg-gray-950 py-12 px-6 pt-28">
      <div className="max-w-6xl mx-auto grid grid-cols-1 lg:grid-cols-12 gap-8">
        
        {/* --- LEFT COLUMN (8 cols) --- */}
        <div className="lg:col-span-8 space-y-8">
          
          {/* Main Card */}
          <div className="bg-gray-900 p-8 rounded-3xl border border-gray-800 shadow-xl relative overflow-hidden">
             <div className="absolute top-0 right-0 w-64 h-64 bg-green-500/5 rounded-full blur-3xl -z-10"></div>

            {/* Title (Cleaned up) */}
            <div className="mb-6">
               <h1 className="text-3xl md:text-4xl font-extrabold text-white leading-tight">{fields.title}</h1>
            </div>

            <div className="flex flex-wrap items-center gap-6 text-sm text-gray-400 mb-8 pb-8 border-b border-gray-800">
                <div className="flex items-center gap-2">
                    <div className="w-8 h-8 rounded-full bg-gradient-to-tr from-green-400 to-blue-500 shadow-lg shadow-green-500/20"></div>
                    <div>
                        <p className="text-[10px] uppercase tracking-wider text-gray-500 font-bold">Creator</p>
                        {fields.owner ? (
                            <Link href={`https://onescan.cc/testnet/account?address=${fields.owner}`} target="_blank" className="text-white hover:text-green-400 font-mono transition-colors">
                                {shortHash(fields.owner)}
                            </Link>
                        ) : "Unknown"}
                    </div>
                </div>
                <div className="hidden sm:block h-8 w-px bg-gray-800"></div>
                <div>
                    <p className="text-[10px] uppercase tracking-wider text-gray-500 font-bold">Launched</p>
                    <p className="text-white">{createdDate}</p>
                </div>
            </div>

            <h3 className="text-lg font-bold text-white mb-3">About this campaign</h3>
            <p className="text-gray-300 text-lg leading-relaxed whitespace-pre-wrap">{fields.description}</p>
          </div>

          {/* Recent Activity Section */}
          <div className="bg-gray-900 p-8 rounded-3xl border border-gray-800">
             <h3 className="text-xl font-bold text-white mb-6">Recent Activity</h3>
             <div className="space-y-4 max-h-[500px] overflow-y-auto pr-2 custom-scrollbar">
                {recentDonations.length > 0 ? (
                    recentDonations.map((e: any) => (
                    <div key={e.id.txDigest} className="flex items-center justify-between p-4 bg-gray-950/50 rounded-xl border border-gray-800 hover:border-gray-700 transition-colors group">
                        <div className="flex items-center gap-4">
                            <div className="w-10 h-10 rounded-full bg-green-900/10 flex items-center justify-center text-green-500 text-xl group-hover:scale-110 transition-transform">üí∏</div>
                            <div>
                                <p className="text-sm text-white font-medium">
                                    <span className="text-gray-500">From</span> <span className="font-mono text-gray-300">{shortHash(e.parsedJson?.donor)}</span>
                                </p>
                                <p className="text-xs text-gray-600">{new Date(Number(e.timestampMs)).toLocaleString()}</p>
                            </div>
                        </div>
                        <div className="text-right">
                            <p className="text-green-400 font-bold text-lg">+{Number(e.parsedJson.amount_donated) / 1e9} OCT</p>
                            <a href={`https://onescan.cc/testnet/transactionBlocksDetail?digest=${e.id.txDigest}`} target="_blank" className="text-xs text-gray-600 hover:text-white hover:underline flex items-center justify-end gap-1">
                                View Tx ‚Üó
                            </a>
                        </div>
                    </div>
                ))
                ) : (
                    <div className="text-center py-10 bg-gray-950/30 rounded-xl border border-gray-800 border-dashed">
                        <p className="text-gray-500">No donations yet. Be the first to support!</p>
                    </div>
                )}
             </div>
          </div>
        </div>

        {/* --- RIGHT COLUMN (4 cols) - Sticky Sidebar --- */}
        <div className="lg:col-span-4 space-y-6">
            
            {/* 1. TOP ACTION BUTTONS (New) */}
            <div className="grid grid-cols-2 gap-4">
                {/* Share Button */}
                <button 
                    onClick={handleShare}
                    className="cursor-pointer flex items-center justify-center gap-2 py-3 px-4 bg-gray-900 hover:bg-gray-800 border border-gray-800 rounded-xl text-gray-300 hover:text-white transition-all group"
                >
                    <span className="text-lg">{copied ? "‚úÖ" : "üîó"}</span>
                    <span className="font-bold text-sm">{copied ? "Copied!" : "Share"}</span>
                </button>

                {/* Verified Badge */}
                <div className="flex items-center justify-center gap-2 py-3 px-4 bg-gray-900 border border-green-900/30 rounded-xl cursor-default">
                    <span className="text-lg">üõ°Ô∏è</span>
                    <span className="font-bold text-sm text-green-400">Verified</span>
                </div>
            </div>

            {/* 2. DONATION CARD (Updated Typography) */}
            <div className="bg-gray-900 p-6 rounded-3xl border border-gray-800 shadow-xl sticky top-28">
                <div className="mb-8 text-center pt-2">
                    <p className="text-xs uppercase tracking-widest text-gray-500 font-bold mb-2">Raised so far</p>
                    
                    {/* Fixed Alignment */}
                    <div className="flex justify-center items-center gap-1">
                        <span className="text-6xl font-black text-white tracking-tighter">
                            {raisedAmount}
                        </span>
                        <span className="text-2xl font-bold text-green-500 ">OCT</span>
                    </div>
                    
                    <div className="w-full bg-gray-800 h-3 rounded-full overflow-hidden mt-6 mb-3">
                        <div className="bg-gradient-to-r from-green-500 to-emerald-400 h-full rounded-full transition-all duration-1000 shadow-[0_0_15px_rgba(34,197,94,0.4)]" style={{ width: `${progressPercent}%` }}></div>
                    </div>
                    <div className="flex justify-between text-xs text-gray-500 font-medium px-1">
                        <span>{progressPercent.toFixed(0)}% Funded</span>
                        <span>Goal: {targetAmount} OCT</span>
                    </div>
                </div>

                <div className="bg-gray-950 p-4 rounded-2xl border border-gray-800 mb-4 focus-within:border-green-500/50 focus-within:ring-1 focus-within:ring-green-500/20 transition-all">
                    <label className="text-[10px] uppercase text-gray-500 font-bold tracking-wider mb-1 block">Donation Amount</label>
                    <div className="relative flex items-center">
                        <input
                            type="number"
                            value={amount}
                            placeholder="0.0"
                            className="w-full bg-transparent text-3xl font-bold text-white placeholder-gray-700 outline-none"
                            onChange={(e) => setAmount(e.target.value)}
                        />
                        <span className="text-gray-500 font-bold text-sm">OCT</span>
                    </div>
                </div>

                <button
                    onClick={donate}
                    disabled={loading}
                    className={`w-full py-4 rounded-xl font-bold text-lg transition-all transform hover:scale-[1.02] active:scale-[0.98] shadow-lg mb-4 ${
                        loading
                        ? "bg-gray-800 text-gray-500 cursor-not-allowed"
                        : "bg-green-600 hover:bg-green-500 text-white shadow-green-900/30"
                    }`}
                >
                    {loading ? (
                        <span className="flex items-center justify-center gap-2">
                            <svg className="animate-spin h-5 w-5 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
                                <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>
                                <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
                            </svg>
                            Processing...
                        </span>
                    ) : "Donate Now"}
                </button>

                <div className="flex items-center gap-2 justify-center text-xs text-gray-500">
                    <svg className="w-3 h-3 text-green-500" fill="currentColor" viewBox="0 0 20 20"><path fillRule="evenodd" d="M5 9V7a5 5 0 0110 0v2a2 2 0 012 2v5a2 2 0 01-2 2H5a2 2 0 01-2-2v-5a2 2 0 012-2zm8-2v2H7V7a3 3 0 016 0z" clipRule="evenodd"></path></svg>
                    <span>Secure on-chain transaction</span>
                </div>
            </div>

            {/* 3. Top Donors Card */}
            <div className="bg-gray-900 p-6 rounded-3xl border border-gray-800">
                <h3 className="text-lg font-bold text-white mb-4 flex items-center gap-2">
                    <span>üèÜ</span> Top Contributors
                </h3>
                <div className="space-y-1">
                    {topDonors.length === 0 ? (
                        <div className="text-center py-6">
                            <p className="text-gray-500 text-sm">Be the first to donate!</p>
                        </div>
                    ) : (
                        topDonors.slice(0, 5).map((d, i) => (
                            <div key={d.donor} className="flex justify-between items-center text-sm p-3 hover:bg-gray-800 rounded-xl transition-colors group cursor-default">
                                <div className="flex items-center gap-3">
                                    <div className={`w-6 h-6 rounded-full flex items-center justify-center text-xs font-bold ${
                                        i === 0 ? "bg-yellow-500/20 text-yellow-500 shadow-[0_0_10px_rgba(234,179,8,0.2)]" : 
                                        i === 1 ? "bg-gray-400/20 text-gray-300" : 
                                        i === 2 ? "bg-orange-500/20 text-orange-500" : "bg-gray-800 text-gray-600"
                                    }`}>
                                        {i + 1}
                                    </div>
                                    <Link href={`https://onescan.cc/testnet/account?address=${d.donor}`} target="_blank" className="font-mono text-gray-400 group-hover:text-green-400 transition-colors">
                                        {shortHash(d.donor)}
                                    </Link>
                                </div>
                                <span className="font-bold text-white group-hover:scale-105 transition-transform">{d.total / 1e9} OCT</span>
                            </div>
                        ))
                    )}
                </div>
            </div>

        </div>
      </div>
    </div>
  );
}
